<div id="ui">

    🏹 Arrows: <span id="arrows">10</span><br>

<script>


const canvas = document.getElementById("c");

const ctx = canvas.getContext("2d");


function resize(){

    canvas.width = innerWidth;

    canvas.height = innerHeight;

}


addEventListener("resize", resize);

resize();


let score = 0;

let combo = 1;

let arrowsLeft = 10;


const ui = {

    score: document.getElementById("score"),

    combo: document.getElementById("combo"),

    arrows: document.getElementById("arrows")

};


const bow = {

    x: 120,

    y: innerHeight * 0.65

};


const cake = {

    x: innerWidth * 0.78,

    y: innerHeight * 0.72,

    w: 160,

    h: 120

};


let dragging = false;

let aimX = 0;

let aimY = 0;


const arrows = [];

const particles = [];

const balloons = [];


let audioCtx = null;


function sound(freq,duration,type="triangle"){


    if(!audioCtx){

        audioCtx = new AudioContext();

    }


    const osc = audioCtx.createOscillator();

    const gain = audioCtx.createGain();


    osc.type = type;

    osc.frequency.value = freq;


    gain.gain.value = 0.06;


    osc.connect(gain);

    gain.connect(audioCtx.destination);


    osc.start();


    gain.gain.exponentialRampToValueAtTime(

        0.0001,

        audioCtx.currentTime + duration

    );


    osc.stop(audioCtx.currentTime + duration);

}


function updateUI(){

    ui.score.textContent = score;

    ui.combo.textContent = "x" + combo;

    ui.arrows.textContent = arrowsLeft;

}


function restart(){


    score = 0;

    combo = 1;

    arrowsLeft = 10;


    arrows.length = 0;

    particles.length = 0;

    balloons.length = 0;


    updateUI();

}


function pointerPos(e){


    if(e.touches){

        return {

            x:e.touches[0].clientX,

            y:e.touches[0].clientY

        };

    }


    return {

        x:e.clientX,

        y:e.clientY

    };

}


function startAim(e){


    if(arrowsLeft <= 0) return;


    dragging = true;


    const p = pointerPos(e);


    aimX = p.x;

    aimY = p.y;

}


function moveAim(e){


    if(!dragging) return;


    const p = pointerPos(e);


    aimX = p.x;

    aimY = p.y;

}


function endAim(){


    if(!dragging || arrowsLeft <= 0) return;


    dragging = false;


    const dx = aimX - bow.x;

    const dy = aimY - bow.y;


    const len = Math.hypot(dx,dy);


    arrows.push({


        x: bow.x,

        y: bow.y,


        vx: dx/len * 13,

        vy: dy/len * 13,


        angle: Math.atan2(dy,dx),

        hit:false


    });


    arrowsLeft--;


    sound(200,0.12);


    updateUI();

}


canvas.addEventListener("mousedown",startAim);

canvas.addEventListener("mousemove",moveAim);

window.addEventListener("mouseup",endAim);


canvas.addEventListener("touchstart",startAim);

canvas.addEventListener("touchmove",moveAim);

window.addEventListener("touchend",endAim);


function confetti(x,y){


    for(let i=0;i<80;i++){


        particles.push({


            x,

            y,


            vx:(Math.random()-0.5)*10,

            vy:(Math.random()-0.5)*10,


            life:80

        });

    }

}


function balloon(){


    balloons.push({


        x:cake.x,

        y:cake.y-120,

        vy:-1.3

    });

}


function drawBackground(){


    const g = ctx.createLinearGradient(

        0,0,0,canvas.height

    );


    g.addColorStop(0,"#314c78");

    g.addColorStop(1,"#d1a067");


    ctx.fillStyle = g;

    ctx.fillRect(0,0,canvas.width,canvas.height);


    ctx.fillStyle="#777";


    ctx.fillRect(

        canvas.width*0.58,

        canvas.height*0.18,

        canvas.width*0.42,

        canvas.height*0.5

    );


    for(let i=0;i<9;i++){


        ctx.fillRect(


            canvas.width*0.58 + i*70,


            canvas.height*0.12,


            45,


            110

        );

    }


    ctx.fillStyle="#4b7a37";


    ctx.fillRect(


        0,

        canvas.height*0.82,

        canvas.width,

        canvas.height

    );

}


function drawCake(){


    ctx.fillStyle="#c79b68";


    ctx.fillRect(


        cake.x-cake.w/2,


        cake.y-cake.h/2,


        cake.w,


        cake.h

    );


    ctx.fillStyle="#f3d6e3";


    ctx.fillRect(


        cake.x-cake.w/2+18,


        cake.y-cake.h/2+18,


        cake.w-36,


        cake.h/2

    );


    for(let i=-1;i<=1;i++){


        const cx = cake.x + i*30;


        ctx.fillStyle="#ffe36e";


        ctx.fillRect(

            cx-3,

            cake.y-cake.h/2-18,

            6,

            18

        );


        const t = performance.now()/200;


        ctx.beginPath();


        ctx.ellipse(


            cx,


            cake.y-cake.h/2-24,


            5,


            10 + Math.sin(t+i)*2,


            0,


            0,


            Math.PI*2

        );


        ctx.fillStyle="orange";


        ctx.fill();

    }

}


function drawWizard(x,y){


    const s = 5;


    ctx.fillStyle="#2b2152";


    ctx.fillRect(x,y,8*s,12*s);


    ctx.fillStyle="#583ca1";


    ctx.beginPath();


    ctx.moveTo(x+4*s,y-10*s);


    ctx.lineTo(x,y);


    ctx.lineTo(x+8*s,y);


    ctx.fill();


    ctx.fillStyle="#f1d1b0";


    ctx.fillRect(x+2*s,y+2*s,4*s,3*s);


    ctx.strokeStyle="#6f4b1d";


    ctx.lineWidth=3;


    ctx.beginPath();


    ctx.moveTo(x+10*s,y);


    ctx.lineTo(x+10*s,y+14*s);


    ctx.stroke();

}


function drawArrow(a){


    ctx.save();


    ctx.translate(a.x,a.y);


    ctx.rotate(a.angle);


    ctx.lineWidth=4;


    ctx.strokeStyle="#3b2413";


    ctx.beginPath();


    ctx.moveTo(-18,0);


    ctx.lineTo(18,0);


    ctx.stroke();


    ctx.beginPath();


    ctx.moveTo(18,0);


    ctx.lineTo(8,-6);


    ctx.lineTo(8,6);


    ctx.closePath();


    ctx.fillStyle="silver";


    ctx.fill();


    ctx.restore();

}


function update(){


    requestAnimationFrame(update);


    drawBackground();


    drawCake();


    drawWizard(

        cake.x+120,

        cake.y-20

    );


    if(dragging){


        ctx.strokeStyle="white";


        ctx.setLineDash([10,10]);


        ctx.beginPath();


        ctx.moveTo(

            bow.x,

            bow.y

        );


        ctx.lineTo(

            aimX,

            aimY

        );


        ctx.stroke();


        ctx.setLineDash([]);

    }


    ctx.beginPath();


    ctx.arc(

        bow.x,

        bow.y,

        35,

        -Math.PI/2,

        Math.PI/2

    );


    ctx.lineWidth=6;


    ctx.strokeStyle="#7c532d";


    ctx.stroke();


    for(const a of arrows){


        if(!a.hit){


            a.x += a.vx;


            a.y += a.vy;


            a.vy += 0.12;


            a.angle = Math.atan2(

                a.vy,

                a.vx

            );


            const hit =


                a.x > cake.x-cake.w/2 &&

                a.x < cake.x+cake.w/2 &&

                a.y > cake.y-cake.h/2 &&

                a.y < cake.y+cake.h/2;


            if(hit){


                a.hit = true;


                score += 100 * combo;


                combo++;


                confetti(

                    cake.x,

                    cake.y

                );


                balloon();


                sound(520,0.3);

                sound(700,0.3);


                updateUI();

            }


            if(a.x > canvas.width ||

               a.y > canvas.height){


                combo = 1;


                updateUI();

            }

        }


        drawArrow(a);

    }


    for(let i=particles.length-1;i>=0;i--){


        const p = particles[i];


        p.x += p.vx;

        p.y += p.vy;


        p.vy += 0.1;


        p.life--;


        ctx.fillStyle = `hsl(${p.life*4},90%,60%)`;


        ctx.fillRect(

            p.x,

            p.y,

            4,

            4

        );


        if(p.life<=0){


            particles.splice(i,1);

        }

    }


    for(let i=balloons.length-1;i>=0;i--){


        const b = balloons[i];


        b.y += b.vy;


        b.x += Math.sin(

            performance.now()/400

        ) * 0.5;


        ctx.beginPath();


        ctx.arc(

            b.x,

            b.y,

            52,

            0,

            Math.PI*2

        );


        ctx.fillStyle="#c92e58";


        ctx.fill();


        ctx.beginPath();


        ctx.moveTo(

            b.x,

            b.y+52

        );


        ctx.lineTo(

            b.x,

            b.y+110

        );


        ctx.strokeStyle="white";


        ctx.stroke();


        ctx.fillStyle="white";


        ctx.font="bold 18px Georgia";


        ctx.textAlign="center";


        ctx.fillText(

            "Happy birthday",

            b.x,

            b.y-8

        );


        ctx.fillText(

            "Dan!",

            b.x,

            b.y+18

        );


        if(b.y < -120){


            balloons.splice(i,1);

        }

    }


    if(arrowsLeft===0 && arrows.every(a =>


        a.x > canvas.width ||

        a.y > canvas.height ||

        a.hit


    )){


        ctx.fillStyle="rgba(0,0,0,.75)";


        ctx.fillRect(

            0,

            0,

            canvas.width,

            canvas.height

        );


        ctx.fillStyle="white";


        ctx.font="bold 48px Georgia";


        ctx.textAlign="center";


        ctx.fillText(


            "Tournament Complete",


            canvas.width/2,


            canvas.height/2 - 40

        );


        ctx.font="32px Georgia";


        ctx.fillText(


            "Final score: " + score,


            canvas.width/2,


            canvas.height/2 + 20

        );


        ctx.fillText(


            "Happy birthday Dan!",


            canvas.width/2,


            canvas.height/2 + 80

        );

    }

}


updateUI();

update();


</script>
